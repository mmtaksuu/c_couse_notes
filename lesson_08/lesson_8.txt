	//scanf("%d", &x);

	while ((c = getchar()) != '\n')
		if (c >= '0' && c <= '9')

		else
			x = x * 16 + c - 'A' + 10;

{
	int x = 25;

	printf("x = %d\n", x);
	++x;
	printf("x = %d\n", x);
	x++;
	printf("x = %d\n", x);
	x = x + 1; //kullanma
	printf("x = %d\n", x);
	x += 1; //kullanma
